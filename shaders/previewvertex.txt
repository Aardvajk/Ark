matrix world;
matrix viewproj;

vector light;

struct VS_INPUT
{
    vector position : POSITION;
    float3 normal   : NORMAL;
    vector color    : COLOR;
    float2 texcoords: TEXCOORD;
};

struct VS_OUTPUT
{
    vector position : POSITION;
	vector color    : COLOR;
    float2 texcoords: TEXCOORD;
};

VS_OUTPUT main(VS_INPUT input)
{
    VS_OUTPUT output = (VS_OUTPUT)0;

    vector p = mul(input.position, world);

    float3 normal = mul(input.normal, (float3x3)world);
    normal = normalize(normal);

    float3 tolight = normalize((float3)light);

    float s = dot(normal, tolight);
    if(s < 0.3f) s = 0.3f;

    output.position = mul(p, viewproj);
    output.color = input.color * s;
    output.texcoords = input.texcoords;
	
    return output;
}
